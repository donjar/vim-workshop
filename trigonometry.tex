  \begin{align*}
    \cos x \cos y &= \frac{1}{2} [2 \cos x \cos y] \\
    &= \frac{1}{2} [(\cos x \cos y - \sin x \sin y) + (\cos x \cos y + \sin x \sin y)] \\
    &= \frac{1}{2} [\cos (x + y) + \cos (x - y)]
  \end{align*}
